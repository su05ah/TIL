##Git 명령어

--설정
git config --global user.name"Your name"
git config --global user.email"Your email address"
--저장소 복제
git clone<저장소 url>
--새로운 원격 저장소 추가
git remote add<원격 저장소><저장소 url>

git add<파일> #새로운 파일 추가
git commmit-m"<메시지>" 
git checkout HEAD<파일>
git reset HEAD<파일>
git branch
git checkout
git merge
git log
git diff
